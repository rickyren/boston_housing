def performance_metric2(y_true, y_predict):
    """"""
    
    n = len(y_true)
    assert len(y_predict) == n
    
    y_bar = np.mean(y_true)
    ss_tot = sum([(y_true[i] - y_bar) ** 2 for i in range(n)])
    ss_res = sum([(y_true[i] - y_predict[i]) ** 2 for i in range(n)])
    score = 1 - ss_res / ss_tot

    return score

def fit_model(X, y):
    """  [X,y]"""
    
    from sklearn.model_selection import KFold, GridSearchCV
    from sklearn.tree import DecisionTreeRegressor
    from sklearn.metrics import make_scorer
    
    cross_validator = KFold()
    
    regressor = DecisionTreeRegressor()

    params = {"max_depth": range(1, 11)}

    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(regressor, params, scoring_fnc, cv = cross_validator)

    #  [X,y]
    grid = grid.fit(X, y)

    # 
    return grid.best_estimator_

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    
    scores = np.zeros(cv)
    m = len(y)
    for i in range(cv):
        train_pos = [x for x in range(m) if x % cv != i]
        val_pos = [x for x in range(m) if x % cv == i]
        estimator.fit(X.iloc[train_pos], y.iloc[train_pos])
        scores[i] = scoring(y.iloc[val_pos], estimator.predict(X.iloc[val_pos]))
        
    return scores

def fit_model2(X, y):
    """  [X,y]"""
    from sklearn.tree import DecisionTreeRegressor
    
    best_score = -100
    for param in range(1,11):
        clf = DecisionTreeRegressor(max_depth = param)
        score = np.mean(cross_val_score(clf, X, y))
        if score > best_score:
            best_score = score
            best_estimator = clf
    
    #
    
    return best_estimator
